2 @RestController 기본

# DRF
from rest_framework.viewsets import ViewSet
from rest_framework.response import Response

class UserViewSet(ViewSet):
    def list(self, request):
        users = User.objects.all()
        data = UserSerializer(users, many=True).data
        return Response(data)

    def retrieve(self, request, pk):
        user = User.objects.get(pk=pk)
        return Response(UserSerializer(user).data)

    def create(self, request):
        s = UserSerializer(data=request.data)
        s.is_valid(raise_exception=True)
        s.save()
        return Response(s.data, status=201)

# urls.py
router.register(r'users', UserViewSet)
@RestController
@RequestMapping("/api/v1/users")
class UserController(
    private val userService: UserService
) {
    @GetMapping
    fun list(): List<UserResponse> =
        userService.findAll().map(UserResponse::from)

    @GetMapping("/{id}")
    fun get(@PathVariable id: Long): UserResponse =
        UserResponse.from(userService.findById(id))

    @PostMapping
    @ResponseStatus(HttpStatus.CREATED)
    fun create(@Valid @RequestBody req: CreateUserRequest): UserResponse {
        val created = userService.create(req.toCommand())
        return UserResponse.from(created)
    }

    @PatchMapping("/{id}")
    fun update(
        @PathVariable id: Long,
        @Valid @RequestBody req: UpdateUserRequest
    ): UserResponse = UserResponse.from(userService.update(id, req))

    @DeleteMapping("/{id}")
    @ResponseStatus(HttpStatus.NO_CONTENT)
    fun delete(@PathVariable id: Long) = userService.delete(id)
}
@RestController vs @Controller

@RestController = @Controller + @ResponseBody. 리턴값을 자동으로 JSON/XML로 직렬화해 HTTP 응답 body에 담습니다. @Controller는 뷰 이름을 리턴(Thymeleaf 같은 템플릿 렌더링용). REST API는 무조건 @RestController.

5 DTO 패턴

DRF의 Serializer가 "모델 ↔ JSON" 변환을 담당한다면, Spring은 data class를 DTO로 사용하고 Jackson이 자동 변환합니다. Entity를 직접 노출하지 말고 DTO를 쓰는 것이 정석. 

// 요청 DTO
data class CreateOrderRequest(
    @field:NotBlank val productId: String,
    @field:Positive val quantity: Int,
    @field:Email val customerEmail: String
) {
    fun toCommand() = CreateOrderCommand(
        productId = productId,
        quantity = quantity,
        customerEmail = customerEmail
    )
}

// 응답 DTO
data class OrderResponse(

    companion object {

            id = order.id,
            productId = order.productId,
            quantity = order.quantity,
            status = order.status.name,
            createdAt = order.createdAt

    }
}

주요 검증 어노테이션

어노테이션의미
@NotNullnull 금지
@NotBlanknull/빈 문자열/공백만 금지 (String)
@NotEmptynull/빈 컬렉션 금지
@Size(min, max)문자열/컬렉션 크기 범위

왜 @field: 접두사?

Kotlin data class의 파라미터는 기본적으로 여러 곳(필드, getter, 생성자 파라미터)에 어노테이션이 적용될 수 있어 명시해야 합니다. jakarta.validation은 필드 레벨에서 동작하므로 @field:NotBlank. Java라면 그냥 @NotBlank만 쓰면 됩니다.

@ResponseStatus(HttpStatus.NO_CONTENT)  // 204
fun delete(@PathVariable id: Long) { ... }

// 3. ResponseEntity로 상세 제어
@GetMapping("/{id}")
fun get(@PathVariable id: Long): ResponseEntity<OrderResponse> {
    val order = service.findByIdOrNull(id)
        ?: return ResponseEntity.notFound().build()

    return ResponseEntity.ok()
        .header("X-Request-Id", UUID.randomUUID().toString())
        .cacheControl(CacheControl.maxAge(60, TimeUnit.SECONDS))
        .body(OrderResponse.from(order))
}

// 4. 생성 + Location 헤더
@PostMapping
fun create(@Valid @RequestBody req: CreateOrderRequest): ResponseEntity<OrderResponse> {
    val created = service.create(req.toCommand())
    val location = URI.create("/orders/${created.id}")
    return ResponseEntity.created(location).body(OrderResponse.from(created))
}

8 전역 예외 처리

@ControllerAdvice는 DRF의 exception_handler에 해당. 모든 컨트롤러의 예외를 한 곳에서 처리. 

// 커스텀 예외
class OrderNotFoundException(orderId: Long) :
    RuntimeException("주문을 찾을 수 없습니다: $orderId")

class InsufficientStockException(
    val productId: String,
    val requested: Int,
    val available: Int
) : RuntimeException("재고 부족: requested=$requested, available=$available")

// 에러 응답 포맷
data class ErrorResponse(
    val code: String,
    val message: String,
    val details: Map<String, Any> = emptyMap(),
    val timestamp: Instant = Instant.now(),
    val path: String? = null
)

@RestControllerAdvice
class GlobalExceptionHandler {

    private val logger = LoggerFactory.getLogger(javaClass)

    // 400: validation 실패
    @ExceptionHandler(MethodArgumentNotValidException::class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    fun handleValidation(e: MethodArgumentNotValidException): ErrorResponse {
        val fields = e.bindingResult.fieldErrors.associate {
            it.field to (it.defaultMessage ?: "invalid")
        }
        return ErrorResponse(
            code = "VALIDATION_FAILED",
            message = "입력값이 올바르지 않습니다",
            details = mapOf("fields" to fields)
        )
    }

    // 404
    @ExceptionHandler(OrderNotFoundException::class)
    @ResponseStatus(HttpStatus.NOT_FOUND)
    fun handleOrderNotFound(e: OrderNotFoundException) = ErrorResponse(
        code = "ORDER_NOT_FOUND",
        message = e.message ?: "not found"
    )

    // 409: 비즈니스 충돌
    @ExceptionHandler(InsufficientStockException::class)
    @ResponseStatus(HttpStatus.CONFLICT)
    fun handleStock(e: InsufficientStockException) = ErrorResponse(
        code = "INSUFFICIENT_STOCK",
        message = e.message ?: "stock error",
        details = mapOf(
            "productId" to e.productId,
            "requested" to e.requested,
            "available" to e.available
        )
    )

    // 500: 예상치 못한 에러
    @ExceptionHandler(Exception::class)
    @ResponseStatus(HttpStatus.INTERNAL_SERVER_ERROR)
    fun handleAll(e: Exception): ErrorResponse {
        logger.error("Unexpected error", e)
        return ErrorResponse(
            code = "INTERNAL_ERROR",
            message = "내부 오류가 발생했습니다"
        )
    }
}
에러 응답 설계 원칙
  • 클라이언트가 기계적으로 분기 가능한 code: "ORDER_NOT_FOUND" 같은 상수
  • 사용자에게 보여줄 message: i18n 가능한 문구
  • 디버깅용 details: 필드별 에러, 요청 ID 등
  • 민감 정보 제외: stacktrace, 내부 SQL 등은 노출 금지
  • HTTP status + body의 code 이중화: 4xx/5xx 구분 + 세부 code

9 계층 아키텍처

① Controller HTTP 처리 (DTO ↔ Command/Query)
② DTO / Command 경계 객체 (요청/응답 모델)
③ Service 비즈니스 로직, 트랜잭션
④ Domain / Entity 도메인 모델, 불변 규칙
⑤ Repository 데이터 접근 (JPA, Redis)
⑥ Database PostgreSQL, Redis, etc. 
// Controller: HTTP만 다룸
@RestController
@RequestMapping("/api/v1/orders")
class OrderController(private val orderService: OrderService) {
    @PostMapping
    fun create(@Valid @RequestBody req: CreateOrderRequest) =
        OrderResponse.from(orderService.create(req.toCommand()))
}

// Service: 비즈니스 로직
@Service
@Transactional
class OrderService(
    private val orderRepo: OrderRepository,
    private val stockService: StockService,
    private val events: ApplicationEventPublisher
) {
    fun create(cmd: CreateOrderCommand): Order {
        stockService.reserve(cmd.productId, cmd.quantity)
        val order = Order.create(cmd)
        val saved = orderRepo.save(order)
        events.publishEvent(OrderCreatedEvent(saved.id))
        return saved
    }
}

// Domain: 도메인 규칙 (JPA Entity 또는 순수 data class)
@Entity
class Order(
    @Id @GeneratedValue val id: Long = 0,
    val productId: String,
    val quantity: Int,
    var status: OrderStatus = OrderStatus.PENDING
) {
    fun cancel() {
        require(status == OrderStatus.PENDING) { "이미 진행된 주문은 취소 불가" }
        status = OrderStatus.CANCELLED
    }
}

// Repository: DB 인터페이스
interface OrderRepository : JpaRepository<Order, Long> {
    fun findByProductId(productId: String): List<Order>
}

@RestControllerAdvice
class GlobalExceptionHandler(
    private val messages: MessageSource
) {
    @ExceptionHandler(OrderNotFoundException::class)
    fun handle(e: OrderNotFoundException, locale: Locale): ErrorResponse {
        val msg = messages.getMessage(
            "error.order.not_found",
            arrayOf(e.orderId),
            locale
        )
        return ErrorResponse(code = "ORDER_NOT_FOUND", message = msg)
    }
}

// 클라이언트는 Accept-Language 헤더로 locale 지정
// Accept-Language: ko, en;q=0.9
글로벌 서비스 운영 팁
  • 모든 유저 입력은 UTF-8. DB collation도 utf8mb4 또는 UTF8
  • 시간대: DB엔 UTC로 저장, 표시는 클라이언트 timezone (Asia/Seoul, America/Los_Angeles)
  • 날짜/숫자 포맷: LocalDate vs Instant 구분 명확히. java.time 사용
  • 통화: BigDecimal + 통화 코드 별도. double 금지
  • 지역별 응답: CDN edge에서 locale 기반 분기 (CloudFront + Lambda@Edge)

12 WebFlux vs MVC

측면Spring MVCSpring WebFlux
모델서블릿 (동기, 스레드-per-요청)리액티브 (Reactor, 이벤트 루프)
내장 서버Tomcat / JettyNetty
반환 타입객체, List, PageMono<T>, Flux<T>
러닝커브낮음높음 (리액티브 사고)
쓰임대부분의 서비스높은 동시성 필요, 백프레셔 중요
언제 WebFlux?

대부분은 MVC로 충분합니다. WebFlux는 "수만 동시 접속 + 대부분 IO 대기" 시나리오에 유리(채팅, streaming, 외부 API 호출 많은 게이트웨이). Kotlin 코루틴과도 통합되지만, 팀 전체가 리액티브 사고를 이해하지 못하면 디버깅이 어려워집니다. MVC + Kotlin 코루틴(Spring Web의 suspend 지원, 6.0+)도 충분히 동시성 확보 가능.

Spring Web MVC + 코루틴

@RestController
class UserController(private val service: UserService) {

    @GetMapping("/users/{id}")
    suspend fun get(@PathVariable id: Long): UserResponse {  // suspend 직접 지원
        return UserResponse.from(service.findById(id))
    }

    @GetMapping("/users/stream")
    fun stream(): Flow<UserResponse> = service.streamUsers()
        .map(UserResponse::from)
}

13 예상 면접 질문

Q. @RestController에서 Entity를 직접 반환하면 왜 안 되나요?
A. 3가지 위험: ①보안 — 비밀번호/토큰 등 내부 필드가 JSON에 노출. ②성능 — Lazy 연관관계 접근시 N+1 쿼리 발생. ③결합도 — API 스키마가 DB 스키마에 묶여 컬럼 추가/이름 변경이 breaking change. DTO를 만들어 필요한 필드만 명시적으로 노출하고, Entity → DTO 변환을 Controller 또는 Service 계층에서 수행합니다.
Q. 예외를 던지는 것 vs Result 타입 사용 중 어느 것을 선호하나요?
A. Spring 생태계에선 예외 기반이 관례입니다. @ControllerAdvice로 전역 매핑이 깔끔하고, @Transactional의 롤백과도 잘 맞습니다(기본적으로 RuntimeException에 롤백). 다만 "예상되는 비즈니스 결과"(사용자 이름 중복 등)는 예외보단 Result 또는 sealed class로 표현하는 것도 좋은 선택 — 호출자가 반드시 처리를 고민하게 되니까. 실제로는 혼합: HTTP 경계에선 예외, 서비스 간에는 sealed class.
Q. 글로벌 서비스에서 API 설계 시 고려사항은?
A. ①i18n: Accept-Language 헤더 + MessageSource로 에러/알림 메시지. ②시간대: 서버 저장은 UTC, 응답은 클라이언트 TZ 또는 ISO-8601 w/ offset. ③지역별 데이터: 데이터베이스를 region으로 분리 (GDPR 등 규제). ④CDN: 정적 응답은 edge 캐시, 동적은 Regional API. ⑤버전 관리: /api/v1로 프리픽스 + deprecated 버전 6~12개월 유지. ⑥통화/숫자: 금액은 BigDecimal, 통화 코드 ISO 4217. ⑦언어 중립 식별자: 텍스트 code 대신 enum.
Q. @RequestBody로 받은 DTO 검증이 실패하면 어떤 예외가 발생하나요?
A. @Valid와 함께 사용한 경우 MethodArgumentNotValidException이 던져지고, 기본 응답은 400 Bad Request. BindingResult 파라미터를 추가하면 예외가 던져지지 않고 결과를 직접 확인 가능하지만 실무에선 @ControllerAdvice로 일괄 처리하는 것이 관용. 중첩 객체는 @Valid(또는 Kotlin에선 @field:Valid)로 깊이 검증 필요.
Q. REST API 버전 관리 전략은?
A. 3가지 흔한 방식: ①URL 프리픽스 (/api/v1/users): 가장 보편적. 가시성 좋고 캐시/라우팅 쉬움. ②Content 협상 (Accept: application/vnd.myapi.v2+json): RESTful 순수주의자 선호하지만 디버깅 어려움. ③헤더 (X-API-Version: 2): 간단하지만 표준 덜함. 실무는 URL 프리픽스 + deprecation header 조합이 일반적. 새 버전 출시시 Deprecation: true, Sunset: 2026-06-01 헤더로 단계적 폐기 안내.
핵심 요약
  • @RestController = 컨트롤러 + JSON 자동 직렬화 (Jackson)
  • Entity 직접 노출 금지 → DTO 패턴(요청/응답 data class)
  • @Valid + jakarta.validation으로 검증, 실패는 @ControllerAdvice에서 일괄 처리
  • 계층: Controller → Service → Domain → Repository 책임 분리
  • 글로벌 서비스: i18n(MessageSource) + UTC 저장 + ISO-8601
  • OpenAPI(springdoc)로 자동 문서화 + API 버전 관리는 URL 프리픽스
  • 대부분 Spring MVC + 코루틴으로 충분. WebFlux는 특수 케이스만
