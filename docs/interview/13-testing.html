class ValidationTest {

    @ParameterizedTest
    @ValueSource(strings = ["", " ", "\t", "\n"])
    fun `blank strings should be invalid`(value: String) {
        assertFalse(validator.isValid(value))
    }

    @ParameterizedTest
    @CsvSource(
        "1, 2, 3",
        "10, 20, 30",
        "-1, 1, 0"
    )
    fun `add should return sum`(a: Int, b: Int, expected: Int) {
        assertEquals(expected, Calculator.add(a, b))
    }

    @ParameterizedTest
    @MethodSource("userScenarios")
    fun `user validation`(scenario: UserScenario) {
        val result = validator.validate(scenario.user)
        assertEquals(scenario.expectedValid, result.isValid)
    }

    companion object {
        @JvmStatic
        fun userScenarios() = listOf(
            UserScenario(User(name = "Teddy", age = 30), true),
            UserScenario(User(name = "", age = 30), false),
            UserScenario(User(name = "Teddy", age = -1), false)
        )
    }

    data class UserScenario(val user: User, val expectedValid: Boolean)
